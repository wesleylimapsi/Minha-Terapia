    <script>
        // Modal functionality
        document.addEventListener('DOMContentLoaded', function() {
            const dreamsCard = document.getElementById('dreamsCard');
            const dreamsModal = document.getElementById('dreamsModal');
            const reflectionsCard = document.getElementById('reflectionsCard');
            const reflectionsModal = document.getElementById('reflectionsModal');
            const tasksCard = document.getElementById('tasksCard');
            const tasksModal = document.getElementById('tasksModal');
            const closeButtons = document.querySelectorAll('.close-modal');
            
            dreamsCard.addEventListener('click', function() {
                dreamsModal.classList.remove('hidden');
            });
            
            reflectionsCard.addEventListener('click', function() {
                reflectionsModal.classList.remove('hidden');
            });
            
            tasksCard.addEventListener('click', function() {
                tasksModal.classList.remove('hidden');
            });
            
            closeButtons.forEach(button => {
                button.addEventListener('click', function() {
                    dreamsModal.classList.add('hidden');
                    reflectionsModal.classList.add('hidden');
                    tasksModal.classList.add('hidden');
                });
            });
            
            // Emotion tags functionality
            const emotionTags = document.querySelectorAll('.emotion-tag');
            emotionTags.forEach(tag => {
                tag.addEventListener('click', function() {
                    this.classList.toggle('selected');
                });
            });
            
            // Close modals when clicking outside
            window.addEventListener('click', function(event) {
                if (event.target === dreamsModal) {
                    dreamsModal.classList.add('hidden');
                }
                if (event.target === reflectionsModal) {
                    reflectionsModal.classList.add('hidden');
                }
                if (event.target === tasksModal) {
                    tasksModal.classList.add('hidden');
                }
            });
        });
    </script>
